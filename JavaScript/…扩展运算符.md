> `...` 是es6的特性，可以把数组转换成`,`隔开的序列，用在参数中，原来序列化的参数可以用数组


### 使用例子：

```js
//------------1------------参数
let add = (x, y) => x + y;
let numbers = [3, 45];
console.log(add(...numbers))//48



//------------2------------参数max
//ES5 取数组最大值
console.log(Math.max.apply(this, [654, 233, 727]))；
//ES6 扩展运算符
console.log(Math.max(...[654, 233, 727]))
//相当于
console.log(Math.max(654, 233, 727))



//------------3------------数组push
// ES5  写法  
var arr1 = [1, 2, 3];  
var arr2 = [4, 5, 6];  
Array.prototype.push.apply(arr1, arr2); 
//push方法的参数不能是数组，通过apply方法使用push方法 
// ES6  写法  
let arr1 = [1, 2, 3];  
let arr2 = [4, 5, 6];  
arr1.push(...arr2); 



//------------4------------临时合并数组
var arr1 = ['a', 'b'];  
var arr2 = ['c'];  
var arr3 = ['d', 'e'];  
// ES5 的合并数组  
arr1.concat(arr2, arr3);  
// [ 'a', 'b', 'c', 'd', 'e' ]  
// ES6 的合并数组  
[...arr1, ...arr2, ...arr3]  
// [ 'a', 'b', 'c', 'd', 'e' ] 



//------------5------------分割字符串
[...'hello']  
// [ "h", "e", "l", "l", "o" ] 
//ES5
str.split('')



//------------6------------赋值
const [x,...y] = [1,2,3,4,5,6];
x = 1;
y = [2,3,4,5,6];



//------------7------------map.keys()
let map = new Map([  
[1, 'one'],  
[2, 'two'],  
[3, 'three'],  
]);  
let arr = [...map.keys()]; // [1, 2, 3]



//------------8------------数组去重
var a = [1,1,1,2]
let set = [...new Set(a)]



//------------9------------新建Date
// ES5  
new (Date.bind.apply(Date, [null, 2015, 1, 1]))  
// ES6  
new Date(...[2015, 1, 1]); 



//------------9------------将Iterator转为数组
var nodeList = document.querySelectorAll('div');  
var array = [...nodeList];  
```
